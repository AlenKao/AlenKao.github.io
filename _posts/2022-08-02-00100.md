【[題目敘述]】  
【解題想法】 直接在範圍內計數，隨時記錄最大值。   

程式碼
------
```c++
#include <iostream>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int i, j;
    while (cin >> i >> j) {
        int mx = 0;
        for (int k=min(i, j); k<=max(i, j); k++) {
            int num = k, cnt = 1;
            while (num != 1) {
                if (num % 2) num = 3*num + 1;
                else num /= 2;
                cnt ++;
            }
            mx = max(mx, cnt);
        }
        cout << i << " " << j << " " << mx << "\n";
    }
    return 0;
}
```
