10056 - What is the Probability?
======

【解題想法】   

程式碼
------
```c++
#include <iostream>
#include <iomanip>
#include <cmath>

using namespace std;

int main () {
    ios_base:: sync_with_stdio(false);
    cin.tie(0);

    int S, n, k;
    double p;
    cin >> S;
    while (S--) {
        cin >> n >> p >> k;
        if (p == 0.0) {
            cout << "0.0000\n";
            continue;
        }

        double first = pow((1.0 - p), (k - 1)) * p;
        double r = pow(1.0 - p, n);
        cout << fixed << setprecision(4) << first/(1.0 - r) << "\n";
    }
    return 0;
}
```
